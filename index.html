Seção 3

const arrayElementos = ["Banana", "Morango", "Maçã", "Uva"];
let frutaEscolhida;
for (let index = 0; index < 2; index++) {
 frutaEscolhida = arrayElementos[index];
}
switch (frutaEscolhida) {
 case "Banana":
  console.log("Bolo de banana");
 break;
 case "Morango":
  console.log("Vitamina de morango")
 break;
 case "Maçã":
  console.log("Torta de maçã")
 break;
 case "Uva":
  console.log("Suco de uva")
 break;
 default:
  console.log("Nenhuma fruta foi escolhida");
 break;
}
